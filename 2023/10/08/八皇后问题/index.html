洛谷p1219八皇后
采用按行枚举的方式进行深度搜索
//这是一道典型的八皇后问题，我要输出前三个解，创建几个数组

#include<iostream>
using namespace std;
const int N = 110;
int n,idx,a;
int g[N];
bool dg[N], udg[N], col[N];//col表示列，dg表示对角线，udg表示副对角线，g表示地图
void  dfs(int u)
{
    if (u == n) {
        if (idx < 3) {
            for (int i = 0; i < n; i++) {
                if (g[i] != -1)cout << g[i]+1 << " ";
            }
            cout << endl;
            idx++;
        }
    a++;
    return;
    }
    for (int i = 0; i < n; i++) {
        if (!col[i] && !dg[u + i] && !udg[n - u + i]) {
            g[u] = i;
            col[i] = dg[u + i] = udg[n - u + i] = true;
            dfs(u + 1);

            col[i] = dg[u + i] = udg[n - u + i] = false;
            g[u] = -1;
        }
    }
}
int main() {
    cin >> n;
    idx = 0;
    memset(g, -1, sizeof g);
    dfs(0);
    cout << a;
}```

p1162填色问题
这道题花了两个小时,代码量可以很少,本题思路是把边框联通的0全部标记,将剩下的0变为2
以下是完整的代码块

#include<cstring>
using namespace std;
int n;
const int N = 35;
int g[N][N];
int vis[N][N];

void dfs(int x,int y) {
    int dx[5] = { 0,-1,0,1,0 }, dy[5] = { 0,0,1,0,-1 };
    for (int i = 1; i <= 4; i++) {
        int x1 = x + dx[i], y1 = y + dy[i];
        if (x1<1 || x1>n || y1<1 || y1>n || vis[x1][y1] == 0||g[x1][y1]==1)continue;
        else {
        //	cout << x1 << " " << y1 << endl;
            vis[x1][y1] = 0;
            dfs(x1, y1);
        }

    }
    

}
int main() {
    memset(vis, -1,sizeof vis );
    cin >> n;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++)cin >> g[i][j];
    }
    //分别在左右上下的边框进行遍历并且打标记
    for (int i = 1; i <= n; i++) {
            if (g[1][i] != 1 && vis[1][i] ==-1) {
                dfs(1, i);
            }
            if (g[n][i] != 1 && vis[n][i] == -1) {
                dfs(n, i);
            }
            if (g[i][1] != 1 && vis[i][1] == -1) {
                dfs(i, 1);
            }
            if (g[i][n] != 1 && vis[i][n] == -1) {
                dfs(i, n);
            }
    }
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            if (vis[i][j] == -1 && g[i][j] == 1)cout << 1<<" ";
            if (vis[i][j] == 0 && g[i][j] == 0)cout << 0<<" ";
            if (vis[i][j] == -1 && g[i][j] == 0)cout << 2 <<" ";
            //针对标记进行输出的更改~冗长~
        }cout << endl;
    }


    return 0;
}
